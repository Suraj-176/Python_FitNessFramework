<script>
(function() {
  var pageSize = 10;
  var currentPage = 1;
  var allCards = [];

  function iciGenerateCardView() {
    var wrapper = document.getElementById("ici-contents-wrapper");
    var grid = document.getElementById("ici-suite-grid");
    if (!grid || !wrapper) return;
    grid.innerHTML = "";

    // Find the first outer main UL container compiled by FitNesse! (Flawless DOM detection)
    var mainUL = wrapper.querySelector("ul");
    if (!mainUL) {
      grid.innerHTML = '<div style="grid-column: 1/-1; text-align: center; padding: 40px; color: #64748b; font-weight: bold;">No active suites or test cases found. Click Create above to add one!</div>';
      return;
    }

    var topLevelItems = Array.from(mainUL.children);
    if (topLevelItems.length === 0) {
      grid.innerHTML = '<div style="grid-column: 1/-1; text-align: center; padding: 40px; color: #64748b; font-weight: bold;">No active suites or test cases found. Click Create above to add one!</div>';
      return;
    }

    topLevelItems.forEach(function(li) {
      // Find nested list (which represents nested test cases!)
      var childList = li.querySelector("ul");
      
      // Calculate immediate text content only (ignoring nested list items)
      var immediateText = "";
      var link = li.querySelector("a");
      if (!link) return;

      var originalName = link.textContent.trim();
      var name = originalName.replace(/\s*\*+/, "").replace(/\s*\++/, "").trim();
      var href = link.href;

      // Robust check: look at link class ('suite' or 'test'), name keywords, and child list!
      var nameLower = name.toLowerCase();
      var isSuite = link.className.indexOf("suite") !== -1 || 
                    (link.className.indexOf("test") === -1 && 
                     (originalName.indexOf("*") !== -1 || 
                      childList !== null || 
                      nameLower.indexOf("suite") !== -1 || 
                      nameLower.indexOf("regression") !== -1 || 
                      nameLower.indexOf("smoke") !== -1 || 
                      nameLower.indexOf("sanity") !== -1 || 
                      nameLower.indexOf("demo") !== -1));
      
      var testCount = 0;
      if (childList) {
        testCount = childList.querySelectorAll("li").length;
      }

      // ── Icon Mapping based on Suite Name ──
      var icon = "📁";
      var nameLower = name.toLowerCase();
      if (nameLower.indexOf("smoke") !== -1) icon = "💨";
      else if (nameLower.indexOf("sanity") !== -1) icon = "🛡️";
      else if (nameLower.indexOf("regression") !== -1) icon = "🧪";
      else if (!isSuite) icon = "📄";

      // ── Dynamic Descriptions Mapping based on Suite Name ──
      var desc = "Dynamic corporate API test suite.";
      if (nameLower.indexOf("smoke") !== -1) {
        desc = "Checks critical application functionality to confirm the build is stable for testing.";
      } else if (nameLower.indexOf("sanity") !== -1) {
        desc = "Verifies specific changes or fixes and their immediate impact.";
      } else if (nameLower.indexOf("regression") !== -1) {
        desc = "Validates that previously working features remain unaffected by recent changes.";
      } else if (!isSuite) {
        desc = "Single executable REST API automation test case.";
      }

      // ── Create Suite Card DOM Element ──
      var card = document.createElement("div");
      card.className = "ici-suite-card" + (isSuite ? "" : " test-case-card");

      // Top section (Icon + Title & details)
      var topDiv = document.createElement("div");
      topDiv.className = "ici-card-top";

      var iconSpan = document.createElement("span");
      iconSpan.className = "ici-card-icon";
      iconSpan.textContent = icon;
      topDiv.appendChild(iconSpan);

      var detailsDiv = document.createElement("div");
      detailsDiv.className = "ici-card-details";

      var h3 = document.createElement("h3");
      h3.textContent = name;
      detailsDiv.appendChild(h3);

      var pDesc = document.createElement("p");
      if (isSuite) {
        pDesc.textContent = testCount + (testCount === 1 ? " Test Case" : " Test Cases");
      } else {
        pDesc.textContent = "Single Test Page";
      }
      detailsDiv.appendChild(pDesc);

      topDiv.appendChild(detailsDiv);
      card.appendChild(topDiv);

      // Body Description section
      var descPara = document.createElement("div");
      descPara.className = "ici-card-body-desc";
      descPara.textContent = desc;
      card.appendChild(descPara);

      // Actions section (Run + Open buttons)
      var actionsDiv = document.createElement("div");
      actionsDiv.className = "ici-card-actions";

      var runBtn = document.createElement("a");
      runBtn.className = "ici-card-btn run";
      runBtn.href = href + (isSuite ? "?suite" : "?test");
      runBtn.textContent = "▶ Run";
      actionsDiv.appendChild(runBtn);

      var openBtn = document.createElement("a");
      openBtn.className = "ici-card-btn open";
      openBtn.href = href;
      openBtn.textContent = "📂 Open";
      actionsDiv.appendChild(openBtn);

      card.appendChild(actionsDiv);
      grid.appendChild(card);
    });

    // ── Apply Dynamic Client-Side Pagination (Max 10 cards, No scroll!) ──
    allCards = Array.from(grid.children);
    if (allCards.length > pageSize) {
      setupPaginationControls(grid);
      renderPage(1);
    }
  }

  function renderPage(page) {
    currentPage = page;
    var startIndex = (page - 1) * pageSize;
    var endIndex = startIndex + pageSize;

    // Show only the 10 entries of the current page, hide the rest
    allCards.forEach(function(card, idx) {
      if (idx >= startIndex && idx < endIndex) {
        card.style.display = ""; // Default display
      } else {
        card.style.display = "none"; // Hide
      }
    });

    // Update pagination controls text and button states
    var totalPages = Math.ceil(allCards.length / pageSize);
    document.getElementById("ici-pag-info").textContent = "Page " + page + " of " + totalPages;
    document.getElementById("ici-prev-btn").disabled = (page === 1);
    document.getElementById("ici-next-btn").disabled = (page >= totalPages);
  }

  function setupPaginationControls(parentContainer) {
    if (document.getElementById("ici-pagination-controls")) return;

    var controls = document.createElement("div");
    controls.id = "ici-pagination-controls";
    controls.className = "ici-pagination-controls";

    // Previous Button
    var prevBtn = document.createElement("button");
    prevBtn.id = "ici-prev-btn";
    prevBtn.className = "ici-pag-btn";
    prevBtn.textContent = "◀ Previous";
    prevBtn.onclick = function() {
      if (currentPage > 1) renderPage(currentPage - 1);
    };
    controls.appendChild(prevBtn);

    // Page Info text
    var info = document.createElement("span");
    info.id = "ici-pag-info";
    info.className = "ici-pag-info";
    info.textContent = "Page 1 of 1";
    controls.appendChild(info);

    // Next Button
    var nextBtn = document.createElement("button");
    nextBtn.id = "ici-next-btn";
    nextBtn.className = "ici-pag-btn";
    nextBtn.textContent = "Next ▶";
    nextBtn.onclick = function() {
      var totalPages = Math.ceil(allCards.length / pageSize);
      if (currentPage < totalPages) renderPage(currentPage + 1);
    };
    controls.appendChild(nextBtn);

    // Append directly below the list
    parentContainer.parentNode.appendChild(controls);
  }

  /* ── 2. Standard List Formatter (runs recursively if List View is enabled!) ── */
  function iciFormatPageList() {
    var items = document.querySelectorAll("#ici-contents-wrapper li");
    items.forEach(function(li) {
      var childList = li.querySelector("ul");
      var immediateText = "";
      for (var i = 0; i < li.childNodes.length; i++) {
        var node = li.childNodes[i];
        if (node.nodeType === Node.TEXT_NODE) immediateText += node.textContent;
        else if (node.tagName === "A") immediateText += node.textContent;
      }
      
      // Suite Page Formatter
      if (immediateText.indexOf("*") !== -1 || childList !== null) {
        var a = li.querySelector("a");
        if (a) {
          // If the inline buttons are already appended, skip appending again!
          if (li.querySelector(".ici-list-actions")) return;

          var name = a.textContent.replace(/\s*\*+/, "").trim();
          var testCount = childList ? childList.querySelectorAll("li").length : 0;
          var href = a.href;
          
          a.textContent = "📁 " + name + " (" + testCount + ")";
          a.style.fontWeight = "bold";
          a.style.color = "#004A80";
          a.style.fontSize = "15px";
          if (childList) childList.style.display = "none";

          // Create and append beautiful inline action buttons on the right! 🟢
          var actions = document.createElement("div");
          actions.className = "ici-list-actions";

          var run = document.createElement("a");
          run.className = "ici-list-btn run";
          run.href = href + "?suite";
          run.textContent = "▶ Run";
          actions.appendChild(run);

          var open = document.createElement("a");
          open.className = "ici-list-btn open";
          open.href = href;
          open.textContent = "📂 Open";
          actions.appendChild(open);

          li.appendChild(actions);
        }
        for (var i = 0; i < li.childNodes.length; i++) {
          var node = li.childNodes[i];
          if (node.nodeType === Node.TEXT_NODE && node.textContent.indexOf("*") !== -1) {
            node.textContent = node.textContent.replace(/\s*\*+/, "");
          }
        }
      }
      // Single Test Case Formatter
      else if (immediateText.indexOf("+") !== -1) {
        var a = li.querySelector("a");
        if (a) {
          if (li.querySelector(".ici-list-actions")) return;

          var name = a.textContent.replace(/\s*\++/, "").trim();
          var href = a.href;

          a.textContent = "📄 " + name;
          a.style.color = "#444";
          a.style.fontSize = "14px";
          a.style.fontWeight = "500";

          // Create and append beautiful inline action buttons on the right! 🟢
          var actions = document.createElement("div");
          actions.className = "ici-list-actions";

          var run = document.createElement("a");
          run.className = "ici-list-btn run";
          run.href = href + "?test";
          run.textContent = "▶ Run";
          actions.appendChild(run);

          var open = document.createElement("a");
          open.className = "ici-list-btn open";
          open.href = href;
          open.textContent = "📂 Open";
          actions.appendChild(open);

          li.appendChild(actions);
        }
        for (var i = 0; i < li.childNodes.length; i++) {
          var node = li.childNodes[i];
          if (node.nodeType === Node.TEXT_NODE && node.textContent.indexOf("+") !== -1) {
            node.textContent = node.textContent.replace(/\s*\++/, "");
          }
        }
      }
    });
  }

  /* ── 3. Dynamic Card/List View Toggle Logic! ── */
  window.iciSetView = function(view) {
    var wrapper = document.getElementById("ici-contents-wrapper");
    var grid = document.getElementById("ici-suite-grid");
    var pag = document.getElementById("ici-pagination-controls");
    
    var btnCard = document.getElementById("ici-btn-card");
    var btnList = document.getElementById("ici-btn-list");
    
    if (!wrapper || !grid) return;
    
    if (view === "card") {
      // Show Card view, hide List view
      wrapper.style.setAttribute ? wrapper.style.setAttribute("display", "none", "important") : wrapper.style.cssText = "display: none !important;";
      grid.style.cssText = "display: grid !important;";
      if (pag) pag.style.cssText = "display: flex !important;";
      
      btnCard.className = "ici-toggle-btn active";
      btnList.className = "ici-toggle-btn";
      
      // Re-render and make sure pagination styles/page-sizes apply
      iciGenerateCardView();
    } else {
      // Show formatted List view, hide Card view
      grid.style.cssText = "display: none !important;";
      if (pag) pag.style.cssText = "display: none !important;";
      wrapper.style.cssText = "display: block !important;";
      
      btnList.className = "ici-toggle-btn active";
      btnCard.className = "ici-toggle-btn";
      
      // Format the lists natively with folder/sheet icons and counts!
      iciFormatPageList();
    }
  };

  // Run card generation automatically on load (Card view by default!)
  setTimeout(function() {
    iciGenerateCardView();
    iciSetView("card"); // Force Card View as default on boot!
  }, 100);

})();
</script>
